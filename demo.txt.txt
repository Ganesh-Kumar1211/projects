package demo1;

import java.lang.classfile.instruction.SwitchCase;
import java.util.Scanner;

public class Calculator {
public static int add(int[] len)
{
    int temp=0;
    for (int l:len)
    {
        temp+=l;
    }
    return temp;
}
    public static int sub(int[] len)
    {
        int temp=len[0];
        for (int l=1;l<len.length;l++)
        {
            temp-=len[l];
        }
        return temp;
    }
    public static int multi(int[] len)
    {
        int temp=1;
        for (int l:len)
        {
            temp*=l;
        }
        return temp;
    }
    public static int dive(int[] len) {
        int temp=len[0];
        for(int i=1;i< len.length;i++)
        {
            temp/=len[i];
        }
        return temp;
    }

    public static double mod(int [] len)
    {
        int temp=len[0];
        for(int i=1;i< len.length;i++)
        {
            temp%= len[i];
        }
        return temp;
    }

    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);

        System.out.println("enter your dogits");
        int size= sc.nextInt();

        int[] len=new int[size];

        for(int i=0;i< len.length;i++)
        {
            System.out.println(i+1+"enter your vsl");
            len[i]=sc.nextInt();
        }

        System.out.println("1. Add(+)");
        System.out.println("2.subtract(-)");
        System.out.println("3.multiplication");
        System.out.println("4.division");
        System.out.println("5.modulo");
        System.out.println("enter choice(1 to 4)");


        int choice= sc.nextInt();
        double result;

        switch (choice){

            case 1:
                result = add(len);
                System.out.println("result="+result);
                break;
            case 2:
                result = sub(len);
                System.out.println("result="+result);
                break;
            case 3:
                result = multi(len);
                System.out.println("result="+result);
                break;
            case 4:
                result = dive(len);
              System.out.println("result="+result);
                break;
            case 5:
                result=mod(len);
                System.out.println(result);
                break;
            default:
                System.out.println("invalid syntax");

                sc.close();
        }

    }

}
